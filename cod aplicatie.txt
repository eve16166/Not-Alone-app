@Composable
fun AppNavigation() {

    val stack = remember { mutableStateListOf("welcome") }
    val current = stack.last()

    fun nav(s: String) = stack.add(s)
    fun back() {
        if (stack.size > 1) stack.removeLast()
    }

    when {

        // 🌟 INTRO FLOW
        current == "welcome" -> WelcomeScreen { nav("intro1") }
        current == "intro1" -> IntroScreen1 { nav("intro2") }
        current == "intro2" -> IntroScreen2 { nav("intro3") }
        current == "intro3" -> IntroScreen3 { nav("intro4") }
        current == "intro4" -> IntroScreen4 { nav("register") }

        current == "register" -> RegisterScreen { nav("home") }

        // 🏠 HOME
        current == "home" -> HomeScreen { nav(it) }

        // 👶 KIDS
        current == "kids" -> KidsScreen(
            onStartChat = { name -> nav("chat_$name") },
            onBack = { back() }
        )

        // 💬 CHAT (FIXED)
        current.startsWith("chat_") -> {
            val name = current.removePrefix("chat_")

            ChatScreen(
                name = name,
                onBack = { back() },
                onSend = { nav("working") }
            )
        }

        current == "working" -> WorkingScreen(onBack = { back() })

        // 👨‍👩‍👧 ADULTS
        current == "adultsMenu" -> AdultsMenuScreen(
            onNavigate = { nav(it) },
            onBack = { back() }
        )

        current == "danger_domestic" -> DomesticViolenceScreen { back() }

        current == "danger_general" -> DangerGeneralScreen(
            onClickMap = { nav("unavailable") },
            onBack = { back() }
        )

        current == "danger_car" -> DangerCarScreen(
            onClickMap = { nav("unavailable") },
            onBack = { back() }
        )

        current == "unavailable" -> UnavailableScreen(
            onRefresh = { nav("loading") },
            onBack = { back() }
        )

        current == "loading" -> LoadingScreen(onBack = { back() })

        current == "older" -> OlderHelpScreen { back() }

        current == "safe" -> SafeHavenScreen(
            onBack = { back() },
            onNavigate = { nav(it) }
        )
        current == "safe_detail" -> DangerGeneralScreen(
            onClickMap = { nav("unavailable") },
            onBack = { back() }
        )

        // 🔥 SAFE LOADING
        current.startsWith("safe_loading_") -> {
            SafeLoadingScreen(
                type = current,
                onBack = { back() }
            )
        }

        // ❌ FALLBACK
        else -> {
            Box(
                Modifier.fillMaxSize(),
                contentAlignment = Alignment.Center
            ) {
                Text("Route not found:\n$current", color = Color.Red)
            }
        }
    }
}

data class ChatMessage(
    val text: String,
    val isUser: Boolean
)

@OptIn(ExperimentalMaterial3Api::class)
@Composable
fun ChatScreen(
    name: String,
    onBack: () -> Unit,
    onSend: () -> Unit
) {

    var msg by remember { mutableStateOf("") }
    var messages by remember { mutableStateOf(listOf<ChatMessage>()) }

    val scrollState = rememberScrollState()

    // 🔥 auto scroll
    LaunchedEffect(messages.size) {
        scrollState.animateScrollTo(scrollState.maxValue)
    }

    Scaffold(
        topBar = {
            TopAppBar(
                title = { Text(name) },
                navigationIcon = {
                    IconButton(onClick = onBack) {
                        Icon(Icons.Default.ArrowBack, null)
                    }
                }
            )
        }
    ) { padding ->

        Column(
            modifier = Modifier
                .fillMaxSize()
                .padding(padding)
        ) {

            // 💬 MESSAGES
            Column(
                modifier = Modifier
                    .weight(1f)
                    .verticalScroll(scrollState)
                    .padding(12.dp)
                    .animateContentSize()
            ) {

                messages.forEach { message ->

                    Row(
                        modifier = Modifier.fillMaxWidth(),
                        horizontalArrangement = if (message.isUser)
                            Arrangement.End else Arrangement.Start
                    ) {

                        Card(
                            shape = RoundedCornerShape(16.dp),
                            colors = CardDefaults.cardColors(
                                containerColor = if (message.isUser)
                                    Color(0xFF3B82F6)
                                else
                                    Color(0xFF1E293B)
                            )
                        ) {
                            Text(
                                text = message.text,
                                color = Color.White,
                                modifier = Modifier.padding(12.dp)
                            )
                        }
                    }

                    Spacer(Modifier.height(8.dp))
                }
            }

            // ⬇️ INPUT
            Row(
                modifier = Modifier
                    .fillMaxWidth()
                    .padding(12.dp),
                verticalAlignment = Alignment.CenterVertically
            ) {

                TextField(
                    value = msg,
                    onValueChange = { msg = it },
                    placeholder = { Text("Type a message...") },
                    modifier = Modifier.weight(1f),
                    shape = RoundedCornerShape(24.dp),
                    colors = TextFieldDefaults.colors(
                        focusedContainerColor = Color(0xFF1E293B),
                        unfocusedContainerColor = Color(0xFF1E293B),
                        focusedTextColor = Color.White,
                        unfocusedTextColor = Color.White
                    )
                )

                Spacer(Modifier.width(8.dp))

                Button(
                    onClick = {
                        if (msg.isNotBlank()) {

                            // 🧑 user message
                            messages = messages + ChatMessage(msg, true)

                            // 🤖 bot logic
                            val reply = when {
                                msg.contains("sad", true) ->
                                    "I'm really sorry you're feeling like this 💙"
                                msg.contains("scared", true) ->
                                    "You're safe here. Talk to me."
                                msg.contains("hey", true) ->
                                    "Hey! How are you?"
                                msg.contains("I'm not okey", true) ->
                                    "I'm here. Tell me what happened. I can help you"
                                msg.contains("can't trust", true) ->
                                    "I'm here to help you. You have just to trust me so I can help you."
                                msg.contains("bad ", true) ->
                                    "I'm here. What happened."
                                msg.contains("can you help", true) ->
                                    "Sure! I'm here for you. You can talk with me about anything."
                                msg.contains("argue", true) ->
                                    "I'm here. Tell me what happened."
                                msg.contains("I'm crying", true) ->
                                    "Oh..I'm sorry! Calm down! I'm here for you. You're not alone anymore. Can you tell me what happened."
                                msg.contains("v", true) ->
                                    "I'm here. Tell me what happened."
                                msg.contains("help", true) ->
                                    "I'm here. Tell me what happened."
                                else ->
                                    "I understand. Tell me more."
                            }

                            messages = messages + ChatMessage(reply, false)

                            msg = ""
                        }
                    },
                    shape = RoundedCornerShape(50),
                    modifier = Modifier.size(50.dp)
                ) {
                    Text("➤")
                }
            }

            // 🔥 NEXT
            PrimaryButton("Next") {
                onSend()
            }
        }
    }
}

    var recording by remember { mutableStateOf(false) }

    Scaffold(
        topBar = {
            TopAppBar(

                navigationIcon = {
                    IconButton(onClick = onBack) {

                    }
                }
            )
        }
    ) { padding ->

        AppBackground {

            Column(
                Modifier.padding(padding).padding(16.dp),
                verticalArrangement = Arrangement.spacedBy(14.dp)
            ) {

                Text(
                    "record a sound so your phone can recognize danger",
                    color = Color.White.copy(0.7f)
                )

                Button(
                    onClick = { recording = !recording },
                    colors = ButtonDefaults.buttonColors(
                        containerColor = if (recording) Color.Red else Color.Gray
                    )
                ) {
                    Text(if (recording) "Stop" else "Record")
                }

                Button(
                    onClick = {},
                    colors = ButtonDefaults.buttonColors(containerColor = Color.Red),
                    modifier = Modifier.fillMaxWidth().height(120.dp)
                ) {
                    Text("SOS", fontSize = 26.sp)
                }

                TextField(
                    value = input,
                    onValueChange = { input = it },
                    modifier = Modifier.fillMaxWidth()
                )

                PrimaryButton("Save") {
                    if (input.isNotBlank()) {
                        saved = saved + input
                        input = ""
                    }
                }
                Column {
                    Text("Saved words:", color = Color.White)

                    saved.forEach {
                        AppCard {
                            Text(it, color = Color.White)
                        }
                    }
                }
            }
        }
    }
}

@OptIn(ExperimentalMaterial3Api::class)
@Composable
fun SafeLoadingScreen(type: String, onBack: () -> Unit) {

    var showMessage by remember { mutableStateOf(false) }

    LaunchedEffect(type) {
        kotlinx.coroutines.delay(1500)
        showMessage = true
    }

    val message = when(type) {
        "safe_loading_water" -> "We are getting you water 💧"
        "safe_loading_toilet" -> "We are finding a safe bathroom 🚽"
        "safe_loading_help" -> "We are sending help 🆘"
        else -> "We are trying to help you ❤️"
    }

    Scaffold(
        topBar = {
            TopAppBar(
                title = { Text("Connecting") },
                navigationIcon = {
                    IconButton(onClick = onBack) {
                        Icon(Icons.Default.ArrowBack, null)
                    }
                }
            )
        }
    ) { padding ->

        Box(
            Modifier.fillMaxSize().padding(padding),
            contentAlignment = Alignment.Center
        ) {

            if (!showMessage) {
                Column(horizontalAlignment = Alignment.CenterHorizontally) {
                    CircularProgressIndicator()
                    Spacer(Modifier.height(16.dp))
                    Text("Trying to locate help...", color = Color.White)
                }
            } else {
                Column(horizontalAlignment = Alignment.CenterHorizontally) {
                    Text(
                        message,
                        fontSize = 22.sp,
                        color = Color.White
                    )

                    Spacer(Modifier.height(12.dp))

                    Text(
                        "Stay calm. Help is being prepared.",
                        color = Color.White.copy(0.7f)
                    )
                }
            }
        }
    }
}

fun OlderHelpScreen(onBack: () -> Unit) {

    var input by remember { mutableStateOf("") }
    var saved by remember { mutableStateOf(listOf<String>()) }
    var recording by remember { mutableStateOf(false) }

    Scaffold(
        topBar = {
            TopAppBar(
                title = { Text("Older Help") },
                navigationIcon = {
                    IconButton(onClick = onBack) {
                        Icon(Icons.Default.ArrowBack, null)
                    }
                }
            )
        }
    ) { padding ->
//de rezolvat la fiecare buton apasat pe safe haven sa apra
        //+de terminat kids chat screen si de actualizat faza cu older help
        // ca la safeloading screen adica sa fie acele mesaje si sa scoata sunet sa introduc sunet
        AppBackground {

            Column(
                Modifier.padding(padding).padding(16.dp),
                verticalArrangement = Arrangement.spacedBy(14.dp)
            ) {

                Text(
                    "Save words or sounds so your phone can detect danger",
                    color = Color.White.copy(0.7f)
                )

                // 🔴 RECORD BUTTON
                Button(
                    onClick = { recording = !recording },
                    colors = ButtonDefaults.buttonColors(
                        containerColor = if (recording) Color.Red else Color.Gray
                    )
                ) {
                    Text(if (recording) "Stop Recording" else "Record Sound")
                }

                // 🆘 SOS BUTTON
                Button(
                    onClick = {},
                    colors = ButtonDefaults.buttonColors(containerColor = Color.Red),
                    modifier = Modifier.fillMaxWidth().height(120.dp)
                ) {
                    Text("SOS", fontSize = 26.sp)
                }

                // ✍️ INPUT
                TextField(
                    value = input,
                    onValueChange = { input = it },
                    placeholder = { Text("Add keyword (ex: help)") },
                    modifier = Modifier.fillMaxWidth()
                )

                // 💾 SAVE
                PrimaryButton("Save") {
                    if (input.isNotBlank()) {
                        saved = saved + input
                        input = ""
                    }
                }

                // 📌 AFIȘARE CUVINTE SALVATE (asta lipsea și la tine 🔥)
                Column {
                    Text("Saved words:", color = Color.White)

                    saved.forEach {
                        AppCard {
                            Text(it, color = Color.White)
                        }
                    }
                }
            }
        }
    }
}
